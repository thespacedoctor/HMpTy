174 def match(self, ra1, dec1, ra2, dec2, radius, maxmatch=1, convertToArray=True): 

175 """*Crossmatch two lists of ra/dec points* 

176 

177 This is very efficient for large search angles and large lists. Note, if you need to match against the same points many times, you should use a `Matcher` object 

178 

179 **Key Arguments:** 

180 - ``ra1`` -- list, numpy array or single ra value (first coordinate set) 

181 - ``dec1`` -- list, numpy array or single dec value (first coordinate set - must match ra1 array length) 

182 - ``ra2`` -- list, numpy array or single ra value (second coordinate set) 

183 - ``dec2`` -- list, numpy array or single dec value (second coordinate set - must match ra2 array length) 

184 - ``radius`` -- search radius in degrees. Can be list, numpy array or single value. If list or numpy array must be same length as ra1 array length) 

185 - ``maxmatch`` -- maximum number of matches to return. Set to `0` to match all points. Default *1* (i.e. closest match) 

186 - ``convertToArray`` -- convert the coordinates into an array. Default *True*. Can bypass the conversion check if you are sure coordinates in numpy array 

187 

188 **Return:** 

189 - ``matchIndices1`` -- match indices for list1 (ra1, dec1) 

190 - ``matchIndices2`` -- match indices for list2 (ra2, dec2) 

191 - ``sepDeg`` -- separations between matched corrdinates in degrees. All returned arrays are the same size 

192 

193 

194 **Usage:** 

195 

196 To match 2 lists of corrdinates try something like this: 

197 

198 .. code-block:: python 

199 

200 twoArcsec = 2.0 / 3600. 

201 raList1 = [200.0, 200.0, 200.0, 175.23, 21.36] 

202 decList1 = [24.3, 24.3, 24.3, -28.25, -15.32] 

203 raList2 = [200.0, 200.0, 200.0, 175.23, 55.25] 

204 decList2 = [24.3 + 0.75 * twoArcsec, 24.3 + 0.25 * twoArcsec, 

205 24.3 - 0.33 * twoArcsec, -28.25 + 0.58 * twoArcsec, 75.22] 

206 matchIndices1, matchIndices2, seps = mesh.match( 

207 ra1=raList1, 

208 dec1=decList1, 

209 ra2=raList2, 

210 dec2=decList2, 

211 radius=twoArcsec, 

212 maxmatch=0 

213 ) 

214 

215 for m1, m2, s in zip(matchIndices1, matchIndices2, seps): 

216 print(raList1[m1], decList1[m1], " -> ", s * 3600., " arcsec -> ", raList2[m2], decList2[m2]) 

217 

218 Note from the print statement, you can index the arrays ``raList1``, ``decList1`` with the ``matchIndices1`` array values and ``raList2``, ``decList2`` with the ``matchIndices2`` values. 

219 """ 

220 

221 # CONVERT LISTS AND SINGLE VALUES TO ARRAYS OF FLOATS 

222 ra1 = numpy.array(ra1, dtype='f8', ndmin=1, copy=False) 

223 dec1 = numpy.array(dec1, dtype='f8', ndmin=1, copy=False) 

224 ra2 = numpy.array(ra2, dtype='f8', ndmin=1, copy=False) 

225 dec2 = numpy.array(dec2, dtype='f8', ndmin=1, copy=False) 

226 radius = numpy.array(radius, dtype='f8', ndmin=1, copy=False) 

227 

228 # CHECK ARRAY SIZES MATCH 

229 if (ra1.size != dec1.size 

230 or ra2.size != ra2.size): 

231 stup = (ra1.size, dec1.size, ra2.size, dec2.size) 

232 raise ValueError("ra1 must equal dec1 in size " 

233 "and ra2 must equal dec2 in size, " 

234 "got %d,%d and %d,%d" % stup) 

235 

236 if radius.size != 1 and radius.size != ra2.size: 

237 raise ValueError("radius size (%d) != 1 and" 

238 " != ra2,dec2 size (%d)" % (radius.size, ra2.size)) 

239 

240 # QUICK TRIMMING IN DEC SPACE OF BOTH SETS OF ARRAYS 

241 decMatchIndices2 = (numpy.abs(dec1[:, None] - dec2) < radius).any(0) 

242 decMatchIndices2 = numpy.where(decMatchIndices2)[0] 

243 ra2a = ra2[decMatchIndices2] 

244 dec2a = dec2[decMatchIndices2] 

245 

246 decMatchIndices1 = (numpy.abs(dec2[:, None] - dec1) < radius).any(0) 

247 decMatchIndices1 = numpy.where(decMatchIndices1)[0] 

248 ra1a = ra1[decMatchIndices1] 

249 dec1a = dec1[decMatchIndices1] 

250 

251 # new way using a Matcher 

252 depth = self.depth 

253 matcher = Matcher( 

254 log=self.log, 

255 depth=depth, 

256 ra=ra2a, 

257 dec=dec2a, 

258 convertToArray=convertToArray) 

259 matchIndices1, matchIndices2, seps = matcher.match( 

260 ra=ra1a, 

261 dec=dec1a, 

262 radius=radius, 

263 maxmatch=maxmatch) 

264 

265 matchIndices1 = decMatchIndices1[matchIndices1] 

266 matchIndices2 = decMatchIndices2[matchIndices2] 

267 

268 return matchIndices1, matchIndices2, seps
